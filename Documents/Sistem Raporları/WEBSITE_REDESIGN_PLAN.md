## 📦 BACKEND GEREKSİNİMLERİ

### **Database Schema:**

```typescript
// Website Project
interface WebsiteProject {
  id: number;
  name: string;
  domain: string;
  status: 'draft' | 'published' | 'archived';
  themeId: number;
  createdAt: Date;
  publishedAt?: Date;
}

// Pages
interface Page {
  id: number;
  websiteId: number;
  title: string;
  slug: string;
  content: json;
  metaTitle: string;
  metaDescription: string;
  isPublished: boolean;
  views: number;
}

// Integration
interface Integration {
  id: number;
  websiteId: number;
  platform: 'wordpress' | 'shopify' | 'squarespace' | 'woocommerce';
  isActive: boolean;
  settings: json;
  apiKey?: string;
}

// Analytics
interface AnalyticsEvent {
  id: number;
  websiteId: number;
  eventType: string;
  page: string;
  visitorId: string;
  timestamp: Date;
}
```
